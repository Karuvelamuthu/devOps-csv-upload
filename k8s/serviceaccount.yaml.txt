---
# ServiceAccount with AWS IRSA (IAM Roles for Service Accounts)
apiVersion: v1
kind: ServiceAccount
metadata:
  name: upload-sa
  namespace: default
  labels:
    app: csv-upload
  annotations:
    eks.amazonaws.com/role-arn: arn:aws:iam::123456789012:role/s3-upload-role

---
# Role for accessing ConfigMaps and Pods
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: upload-role
  namespace: default
  labels:
    app: csv-upload
rules:
- apiGroups: [""]
  resources: ["configmaps"]
  verbs: ["get", "list", "watch"]
- apiGroups: [""]
  resources: ["pods"]
  verbs: ["get", "list"]
- apiGroups: [""]
  resources: ["pods/log"]
  verbs: ["get"]

---
# RoleBinding connecting ServiceAccount to Role
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: upload-rolebinding
  namespace: default
  labels:
    app: csv-upload
roleRef:
  apiGroup: rbac.authorization.k8s.io
  kind: Role
  name: upload-role
subjects:
- kind: ServiceAccount
  name: upload-sa
  namespace: default
